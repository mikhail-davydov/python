import sys  # импорт модуля для работы с системными потоками
'----------------------'
# ПОТОК ВВОДА sys.stdin:
# Специальный объект-итератор, куда попадает пользовательский ввод
# Работает как односторонний поток: прочитанные данные удаляются навсегда
# Каждый элемент итератора — строка, введенная пользователем
'----------------------'
for line in sys.stdin:
    # итерирование по всем строкам пользовательского ввода
    # line содержит строку с символом \n в конце
    print(line.strip())  # убираем \n перед выводом
'----------------------'
# ЗАВЕРШЕНИЕ ВВОДА:
# Linux/macOS/PyCharm: Ctrl + D
# Windows: Ctrl + Z, затем Enter
# Wing IDE: правой кнопкой → Send EOF → Enter
'----------------------'
data = [line.strip() for line in sys.stdin]
# чтение всех строк в список (убираем \n)
# компактный способ получить все строки сразу
'----------------------'
data = list(map(str.strip, sys.stdin))
# альтернативный способ через map() — убирает \n из всех строк
'----------------------'
# МЕТОДЫ ЧТЕНИЯ sys.stdin:
sys.stdin.read()
# читает весь ввод как одну многострочную строку
# сохраняет все символы \n
'----------------------'
sys.stdin.readlines()
# читает все строки в список
# сохраняет символы \n в каждой строке
'----------------------'
sys.stdin.readline()
# читает одну строку (с \n в конце)
# аналог input(), но не убирает \n
'----------------------'
# ПОТОК ВЫВОДА sys.stdout:
# Специальный объект, куда направляется весь вывод программы
# По умолчанию print() пишет именно в sys.stdout
'----------------------'
sys.stdout.write(string)
# запись строки в поток вывода
# НЕ добавляет \n автоматически!
# НЕ преобразует типы — только строки!
'----------------------'
sys.stdout.write('Hello\n')
# для перевода строки нужно явно добавить \n
'----------------------'
sys.stdout.write(str(number))
# числа нужно преобразовывать в строки самостоятельно
'----------------------'
# СРАВНЕНИЕ print() И sys.stdout.write():
print('Hello')              # добавляет \n автоматически
sys.stdout.write('Hello\n') # \n нужно добавлять вручную

print(42)                   # автоматически преобразует в строку
sys.stdout.write(str(42))   # преобразование в строку вручную
'----------------------'
# ПРАКТИЧЕСКИЙ ПРИМЕР — дублирование ввода:
import sys

for line in sys.stdin:
    sys.stdout.write(line)  # line уже содержит \n

# или проще:
for line in sys.stdin:
    print(line.strip())     # strip() убирает \n, print() добавляет свой
'----------------------'

'----------------------'
# БЕЗОПАСНАЯ РАБОТА С input():
# input() может вызвать EOFError если в stdin нет данных
try:
    line = input()
except EOFError:
    print("Нет данных для чтения")
'----------------------'
# ОПРЕДЕЛЕНИЕ КОЛИЧЕСТВА СТРОК:

for line in sys.stdin:
    lines.append(line.strip())

count = len(lines)

# ОБРАБОТКА ПУСТЫХ СТРОК:
import sys

for line in sys.stdin:
    line = line.strip()
    if line:  # пропускаем пустые строки
        print(f'Обработка: {line}')
'----------------------'

# ✔️ Чтение данных неизвестного объема
# ✔️ Обработка потока данных построчно
# ✔️ Конкурсное программирование
# ✔️ Системные скрипты, обрабатывающие ввод
# ✔️ Фильтры командной строки (pipe)
'----------------------'
# ОСОБЕННОСТИ И ОГРАНИЧЕНИЯ:
# ✔️ sys.stdin — итератор, можно использовать for/map/list comprehensions
# ✔️ Прочитанные данные удаляются из потока безвозвратно
# ✔️ Строки содержат \n в конце (нужно strip())
# ✔️ Завершение ввода требует специальных комбинаций клавиш

# ❌ Нет автоматического преобразования типов в write()
